\begin{frame}{The physical control parameter is the source strength}
  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
      \begin{block}{Two-layer Poisson problem}
      \[
        \Omega_I=[-1,0],\qquad \Omega_E=[0,1],
      \]
      \[
        -\lambda u_I''=0,\qquad
        -\mu u_E''=2\mu\kappa,
      \]
      \[
        u_I(-1)=u_E(1)=0.
      \]
      \end{block}
    \column{0.44\textwidth}
      \begin{block}{Interpretation}
        \begin{itemize}
          \item \(\lambda,\mu\): material conductivities;
          \item \(\delta=\lambda/\mu\): conductivity ratio;
          \item \(\kappa\): normalized source strength;
          \item source amplitude after division by \(\mu\): \(2\kappa\).
        \end{itemize}
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{The bulk fields are explicit}
  Imposing the exterior boundary conditions gives
  \[
    u(z)=
    \begin{cases}
      a(z+1), & z\in\Omega_I,\\[0.2em]
      b(z-1)+\kappa(1-z^2), & z\in\Omega_E.
    \end{cases}
  \]

  \begin{columns}[T,onlytextwidth]
    \column{0.46\textwidth}
      \begin{block}{Interface traces}
        \[
          u_I(0)=a,\qquad u_E(0)=\kappa-b.
        \]
      \end{block}
    \column{0.50\textwidth}
      \begin{block}{Transmission law}
        \[
          \lambda a=\mu b
          =\beta_{\rm phys}(\kappa-b-a).
        \]
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Elimination gives one scalar fixed-point problem}
  Flux continuity gives \(b=\delta a\), hence
  \[
    \boxed{s=[u]=\kappa-(1+\delta)a.}
  \]
  After normalizing the conductance and heat flux by \(\mu\),
  \[
    \widehat q=\delta a=\betahat(s)s.
  \]
  Therefore
  \begin{equation*}
    \boxed{
    a=F_{\kappa,h}(a)
    =\frac{\kappa\betahat_{\kappa,h}(s)}
    {\delta+(1+\delta)\betahat_{\kappa,h}(s)}.}
  \end{equation*}
  The physical interval moves with the control parameter:
  \[
    0<a<\frac{\kappa}{1+\delta},\qquad 0<s<\kappa.
  \]
\end{frame}

\begin{frame}{The new exponential contact law is exactly solvable}
  Choose
  \[
    \boxed{
    \betahat_{\kappa,h}(s)
    =\frac{\delta}{1+\delta}
    \exp\!\left[2(\kappa-2s)+2h\right].}
  \]
  Define
  \[
    x=\frac{2(1+\delta)a}{\kappa}-1
     =1-\frac{2s}{\kappa}.
  \]
  Since \(\kappa-2s=\kappa x\), the complete steady equation becomes
  \[
    \boxed{x=\tanh(\kappa x+h).}
  \]

  \begin{center}
    \small The source parameter now controls both the moving physical scale and the normalized sigmoid slope.
  \end{center}
\end{frame}

\begin{frame}{The critical source strength is one}
  For \(h=0\), the equation is \(x=\tanh(\kappa x)\).
  \begin{columns}[T,onlytextwidth]
    \column{0.47\textwidth}
      \begin{block}{Root count}
        \[
        \begin{array}{ll}
        0<\kappa\le1: & 1\text{ state},\\
        \kappa>1: & 3\text{ states}.
        \end{array}
        \]
        \[
          \boxed{\kappa_c=1.}
        \]
      \end{block}
    \column{0.49\textwidth}
      \begin{block}{Exact branches}
        \small
        \[
          a_c(\kappa)=\frac{\kappa}{2(1+\delta)},
        \]
        \[
          \kappa(x)=\frac{\artanh x}{x},\qquad
          a(x)=\frac{\kappa(x)(1+x)}{2(1+\delta)}.
        \]
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Stability means attraction of the Picard iteration}
  At a fixed point,
  \[
    m_{\rm FP}
    =\kappa\sech^2(\kappa x+h)
    =\kappa(1-x^2).
  \]
  \begin{columns}[T,onlytextwidth]
    \column{0.47\textwidth}
      \begin{block}{Line convention}
        \begin{itemize}
          \item solid: \(|m_{\rm FP}|<1\), attracting;
          \item dashed: \(|m_{\rm FP}|>1\), repelling.
        \end{itemize}
      \end{block}
    \column{0.49\textwidth}
      \begin{block}{Scope}
        This is stability of the fixed-point algorithm. Physical transient stability requires a separate time-dependent interface model.
      \end{block}
  \end{columns}
\end{frame}

\begin{frame}{The saddle-node trajectory is analytic}
  The fold equations are
  \[
    \kappa_{\rm SN}=\frac{1}{1-x_{\rm SN}^2},
    \qquad
    h=\artanh x_{\rm SN}
      -\frac{x_{\rm SN}}{1-x_{\rm SN}^2}.
  \]
  Near \((\kappa,h)=(1,0)\),
  \[
    h=-\frac{2}{3}x_{\rm SN}^3+O(x_{\rm SN}^5),
    \qquad
    \kappa_{\rm SN}-1=x_{\rm SN}^2+O(x_{\rm SN}^4),
  \]
  hence
  \[
    \boxed{\kappa_{\rm SN}-1
      \sim\left(\frac{3|h|}{2}\right)^{2/3}.}
  \]
\end{frame}
